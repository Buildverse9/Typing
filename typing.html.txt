let currentPassage = '';
let passageChars = [];
let typedIndex = 0;
let mistakes = 0;
let wrongSet = new Set();    // indices that are wrong
let correctSet = new Set();  // indices typed correctly
let timerInterval = null;
let secondsLeft = 600; // 10 minutes
let examStarted = false;
let examFinished = false;

function pickPassage() {
  const idx = Math.floor(Math.random() * PASSAGES.length);
  return PASSAGES[idx].trim();
}

function buildPassageDisplay(passage) {
  const display = document.getElementById('passage-display');
  passageChars = passage.split('');
  display.innerHTML = passageChars.map((ch, i) => {
    const escaped = ch === '<' ? '&lt;' : ch === '>' ? '&gt;' : ch === '&' ? '&amp;' : ch === '\n' ? '<br>' : ch;
    return `<span class="char" id="c${i}">${escaped}</span>`;
  }).join('');
  updateCursor();
}

function updateCursor() {
  document.querySelectorAll('.char.current').forEach(el => el.classList.remove('current'));
  const el = document.getElementById('c' + typedIndex);
  if (el) el.classList.add('current');
}

function resetExam() {
  clearInterval(timerInterval);
  timerInterval = null;
  examStarted = false;
  examFinished = false;
  typedIndex = 0;
  mistakes = 0;
  wrongSet.clear();
  correctSet.clear();
  secondsLeft = 600;

  currentPassage = pickPassage();
  buildPassageDisplay(currentPassage);

  document.getElementById('typing-input').value = '';
  document.getElementById('typing-input').disabled = true;
  document.getElementById('timer-val').textContent = '10:00';
  document.getElementById('timer-val').classList.remove('urgent');
  document.getElementById('words-typed').textContent = '0';
  document.getElementById('mistake-count').textContent = '0';
  document.getElementById('accuracy-val').textContent = '100%';
  document.getElementById('net-score').textContent = '—';
  document.getElementById('progress-bar').style.width = '0%';
  document.getElementById('btn-start').textContent = '▶ Start Examination';
  document.getElementById('result-overlay').classList.remove('show');
}

function startExam() {
  // Hide start overlay if present
  const startOverlay = document.getElementById('start-overlay');
  if (startOverlay) startOverlay.style.display = 'none';

  if (examStarted && !examFinished) return;
  if (examFinished) { resetExam(); return; }

  if (!currentPassage) {
    currentPassage = pickPassage();
    buildPassageDisplay(currentPassage);
  }

  examStarted = true;
  document.getElementById('typing-input').disabled = false;
  document.getElementById('typing-input').focus();
  document.getElementById('btn-start').textContent = '⏳ Exam in Progress';

  timerInterval = setInterval(() => {
    secondsLeft--;
    const m = Math.floor(secondsLeft / 60).toString().padStart(2, '0');
    const s = (secondsLeft % 60).toString().padStart(2, '0');
    document.getElementById('timer-val').textContent = `${m}:${s}`;

    if (secondsLeft <= 60) {
      document.getElementById('timer-val').classList.add('urgent');
    }

    if (secondsLeft <= 0) {
      endExam();
    }
  }, 1000);
}

document.getElementById('typing-input').addEventListener('keydown', function(e) {
  if (!examStarted || examFinished) return;

  // Disable navigation keys
  if (['ArrowLeft','ArrowRight','ArrowUp','ArrowDown','Home','End'].includes(e.key)) {
    e.preventDefault();
  }
});

document.getElementById('typing-input').addEventListener('input', function(e) {
  if (!examStarted || examFinished) return;

  const inputVal = this.value;
  const inputLen = inputVal.length;

  // We track character by character
  // The input box holds everything typed so far
  // We only look at the latest character typed

  if (inputLen > passageChars.length) {
    this.value = inputVal.slice(0, passageChars.length);
    return;
  }

  // Update character states
  for (let i = 0; i < inputLen; i++) {
    const el = document.getElementById('c' + i);
    if (!el) continue;
    el.classList.remove('correct', 'wrong', 'current');

    if (inputVal[i] === passageChars[i]) {
      el.classList.add('correct');
      correctSet.add(i);
      wrongSet.delete(i);
    } else {
      el.classList.add('wrong');
      wrongSet.add(i);
      correctSet.delete(i);
    }
  }

  // Reset chars beyond input
  for (let i = inputLen; i < passageChars.length; i++) {
    const el = document.getElementById('c' + i);
    if (el) el.classList.remove('correct', 'wrong', 'current');
  }

  typedIndex = inputLen;
  updateCursor();

  // Count mistakes as total wrong chars
  mistakes = wrongSet.size;

  // Stats
  const totalTyped = inputLen;
  const accuracy = totalTyped > 0 ? Math.round((correctSet.size / totalTyped) * 100) : 100;

  // Words correctly typed = number of complete correct words
  const wordsTyped = countCorrectWords(inputVal);
  const passageWords = currentPassage.trim().split(/\s+/).length;
  const wordsRemaining = Math.max(0, passageWords - countWordsTyped(inputVal));
  const penalty = Math.floor(mistakes / 4);
  const netScore = Math.max(0, wordsTyped - penalty - wordsRemaining);

  document.getElementById('words-typed').textContent = countWordsTyped(inputVal);
  document.getElementById('mistake-count').textContent = mistakes;
  document.getElementById('accuracy-val').textContent = accuracy + '%';
  document.getElementById('net-score').textContent = netScore;

  // Progress
  const progress = (inputLen / passageChars.length) * 100;
  document.getElementById('progress-bar').style.width = progress + '%';

  // Check if passage complete
  if (inputLen >= passageChars.length) {
    endExam();
  }
});

function countWordsTyped(input) {
  return input.trim() === '' ? 0 : input.trim().split(/\s+/).length;
}

function countCorrectWords(input) {
  const passageWords = currentPassage.trim().split(/\s+/);
  const inputWords = input.split(/\s+/);
  let correct = 0;
  for (let i = 0; i < Math.min(inputWords.length, passageWords.length); i++) {
    if (inputWords[i] === passageWords[i]) correct++;
  }
  return correct;
}

function endExam() {
  if (examFinished) return;
  examFinished = true;
  clearInterval(timerInterval);
  document.getElementById('typing-input').disabled = true;
  document.getElementById('btn-start').textContent = '✓ Exam Complete';

  const inputVal = document.getElementById('typing-input').value;
  const passageWords = currentPassage.trim().split(/\s+/);
  const totalPassageWords = passageWords.length;

  const wordsTypedCount = countWordsTyped(inputVal);
  const correctWords = countCorrectWords(inputVal);
  const wordsRemaining = Math.max(0, totalPassageWords - wordsTypedCount);
  const penalty = Math.floor(mistakes / 4);
  const netScore = Math.max(0, correctWords - penalty - wordsRemaining);
  const accuracy = wordsTypedCount > 0
    ? Math.round((correctWords / wordsTypedCount) * 100)
    : 0;

  // Show result
  document.getElementById('res-score').textContent = netScore;
  document.getElementById('res-typed').textContent = wordsTypedCount;
  document.getElementById('res-remaining').textContent = wordsRemaining;
  document.getElementById('res-mistakes').textContent = mistakes;
  document.getElementById('res-accuracy').textContent = accuracy + '%';
  document.getElementById('res-gross').textContent = correctWords;
  document.getElementById('res-penalty').textContent = penalty;

  const verdict = document.getElementById('res-verdict');
  // Typical BHC passing: 30 wpm net = 300/10min words? 
  // Let's say passing = net score >= 30
  if (netScore >= 30) {
    verdict.textContent = '✓ Satisfactory Performance';
    verdict.className = 'result-verdict pass';
  } else {
    verdict.textContent = '✗ Below Passing Standard';
    verdict.className = 'result-verdict fail';
  }

  setTimeout(() => {
    document.getElementById('result-overlay').classList.add('show');
  }, 400);
}
